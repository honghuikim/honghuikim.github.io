% section header: small, near-black, uppercase, thin full-width hairline
\newcommand{\cvsection}[1]{%
  \vspace{13pt}\par\noindent
  {\small\bfseries\color{ink}\MakeUppercase{#1}}\par\vspace{3pt}%
  {\color{rulec}\hrule height 0.6pt}\vspace{8pt}}

% subsection: small italic gray label within a section
\newcommand{\cvsubsection}[1]{%
  \vspace{9pt}\par\noindent{\small\itshape\color{subtext}#1}\par\vspace{5pt}}

\newcommand{\hlink}[2]{\href{#2}{\color{accent}#1}}

% timeline row: left meta (right-aligned, gray, hugs rule) | hairline | content
% left + right first lines share the same font size so their baselines align.
% Row spacing lives INSIDE the right minipage (trailing \vspace) and interline
% glue is suppressed, so the dividing rule runs continuously across all rows.
\newcommand{\trow}[3][6pt]{%
  \noindent\begin{minipage}[t]{\datecol}\raggedleft\strut\color{subtext}#2\end{minipage}%
  \hspace{0.10in}{\color{rulec}\vrule width 0.7pt}\hspace{\dimexpr\gutter-0.10in}%
  \begin{minipage}[t]{\dimexpr\linewidth-\datecol-\gutter-0.7pt}\raggedright\strut #3\par\vspace{#1}\end{minipage}%
  \par\nointerlineskip}
